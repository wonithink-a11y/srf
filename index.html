📥 PC → 모바일 가져오기

      

        PC 앱을 열고 브라우저 콘솔(F12)에서
아래 코드를 실행 → srf_export.json 다운로드
      

      

        
const fixed=['_siloStock_v5','_usageData_v5','_usageDataBySilo_v5','_schedSettings_v5'];
const rawdb=Object.keys(localStorage).filter(k=>/^_rawdb_\d{4}$/.test(k));
const out={};
[...fixed,...rawdb].forEach(k=>{const v=localStorage.getItem(k);if(v)out[k]=JSON.parse(v);});
const a=document.createElement('a');
a.href='data:application/json,'+encodeURIComponent(JSON.stringify(out));
a.download='srf_export.json';
a.click();
